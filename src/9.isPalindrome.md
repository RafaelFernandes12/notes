function isPalindrome(x: number): boolean {
    // Guarda o valor de x, para comparar com res mais tarde
    let y = x
    // res é uma variável de auxílio, ela será x invertida
    let res = 0
    while (x > 0){
        // Pega o ultimo valor de x, 121 = 1; 10 = 0
        const last = x % 10
        // O * 10 serve para manter a casa do número em ordem.
        // 121: 1 + 0 * 10 = 1
        // 12: 2 + 1 * 10 = 12
        // 1: 1 + 12 * 10 = 121
        res = last + res * 10
        // Vai reduzindo as casa de x
        x = Math.floor(x / 10)
    }
    return y === res
};
isPalindrome(121)
isPalindrome(-121)
isPalindrome(10)
export {}
